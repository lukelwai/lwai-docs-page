python拆分pdf

import fitz
from os.path import abspath, dirname
import os


def extract_toc(pdf_path):
    # 打开PDF文件
    doc = fitz.open(pdf_path)

    # 获取目录（书签）
    toc = doc.get_toc()

    # 关闭PDF文件
    doc.close()

    return toc


def split_pdf_by_toc(pdf_path, output_dir):
    # 打开PDF文件
    doc = fitz.open(pdf_path)

    # 获取目录（书签）
    toc = extract_toc(pdf_path)

    if not toc:
        print("未找到目录（书签），无法拆分PDF。")
        return

    # 过滤出一级目录项
    first_level_toc = [item for item in toc if item[0] == 1]

    if not first_level_toc:
        print("未找到一级目录项，无法拆分PDF。")
        return

    # 遍历每个一级目录项
    for i in range(len(first_level_toc) - 1):
        level, title, page_num = first_level_toc[i]
        next_level, next_title, next_page_num = first_level_toc[i + 1]

        # 创建一个新的PDF文档
        new_doc = fitz.open()

        # 添加页面到新的PDF文档
        for page_num in range(page_num - 1, next_page_num - 1):
            new_doc.insert_pdf(doc, from_page=page_num, to_page=page_num)

        # 保存当前章节的PDF文件
        output_path = f"{output_dir}/chapter_{i + 1}_{title}.pdf"
        new_doc.save(output_path)
        new_doc.close()
        print(f"已保存章节: {output_path}")

    # 处理最后一个一级目录项
    level, title, page_num = first_level_toc[-1]
    new_doc = fitz.open()
    for page_num in range(page_num - 1, doc.page_count):
        new_doc.insert_pdf(doc, from_page=page_num, to_page=page_num)

    output_path = f"{output_dir}/chapter_{len(first_level_toc)}_{title}.pdf"
    new_doc.save(output_path)
    new_doc.close()
    print(f"已保存章节: {output_path}")

    # 关闭原始PDF文件
    doc.close()

if __name__ == '__main__':

    # 示例用法
    pdf_path = '信息系统项目管理师第四版-带标签-可搜索.pdf'
    # 获取当前目录，进行输出
    output_dir = abspath(dirname(__file__))

    # 确保输出目录存在（不存在会创建）
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)

    split_pdf_by_toc(pdf_path, output_dir)
