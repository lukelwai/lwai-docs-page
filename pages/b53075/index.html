python 爬虫

在爬取数据之前，我也没想着这东西有多难，只不过是有些陌生不太了解，好家伙就是吃了没见识的亏，以至于我使用抓取静态页面的方式跟动态页面干上好久无果，冷静下来看代码逻辑才发现自己的问题，只不过时间却是被浪费了。

# 1. 注意事项

  1. 爬取页面是动态的还是静态的
    1. 如果是动态页面需要使用特殊的组件帮忙捕抓动态页面比如：selenium
  2. 开发方式：简单的脚本开发、爬虫框架scrapy集成开发
  3. 是否需要进行登录验证
  4. 爬取到的具体数据定位方式xpath、css，具体数据位置
  5. 爬到的数据是否进行入库操作或者写入文件操作
    1. 如果使用框架，需要进行数据处理的会用到管道

# 2. 处理中间件，只修改下载中间件即可SpiderDemoDownloaderMiddleware

# Define here the models for your spider middleware
#
# See documentation in:
# https://docs.scrapy.org/en/latest/topics/spider-middleware.html
import scrapy.http
from scrapy import signals
from selenium import webdriver
import time


class SpiderDemoDownloaderMiddleware:
    # Not all methods need to be defined. If a method is not defined,
    # scrapy acts as if the downloader middleware does not modify the
    # passed objects.

    def __init__(self):
        # 后台启动浏览程序
        options = webdriver.ChromeOptions()
        options.add_argument('--headless')
        self.driver = webdriver.Chrome(options=options)

    @classmethod
    def from_crawler(cls, crawler):
        # This method is used by Scrapy to create your spiders.
        s = cls()
        crawler.signals.connect(s.spider_opened, signal=signals.spider_opened)
        # 下面这一行需要手动添加，作用是爬虫关闭的时候调用关闭模拟浏览器的函数
        crawler.signals.connect(s.spider_closed, signal=signals.spider_closed)
        return s

    def process_request(self, request, spider):
        # Called for each request that goes through the downloader
        # middleware.
        # Must either:
        # - return None: continue processing this request
        # - or return a Response object
        # - or return a Request object
        # - or raise IgnoreRequest: process_exception() methods of
        #   installed downloader middleware will be called
        self.driver.get(request.url)  # 使用浏览器打开请求的URL
        if request.url.find("login") > -1:
            # 输入账号密码进行登录
            # 找到用户名和密码输入框
            inputs = self.driver.find_elements_by_class_name('el-input__inner')  # 根据实际页面元素定位

            # 输入账号密码
            inputs[0].send_keys(spider.settings.get('USER'))
            inputs[1].send_keys(spider.settings.get('PASSWORD'))

            # 找到登录按钮，点击登录
            login_button = self.driver.find_elements_by_class_name("el-button")  # 根据实际页面元素定位
            login_button[0].click()
            return None
        else:
            # 需要等待页面渲染完成
            time.sleep(2)
            # 如果获取网页源码有问题的话，也可以使用这种方式获取指定内容的HTML放到body里面
            # content = self.driver.find_element_by_id('app').get_attribute("outerHTML")
            content = self.driver.page_source  # 获取网页HTML源码
            return scrapy.http.HtmlResponse(url=self.driver.current_url, body=content, encoding='utf-8',
                                            request=request)


    def process_response(self, request, response, spider):
        # Called with the response returned from the downloader.

        # Must either;
        # - return a Response object
        # - return a Request object
        # - or raise IgnoreRequest
        return response

    def process_exception(self, request, exception, spider):
        # Called when a download handler or a process_request()
        # (from other downloader middleware) raises an exception.

        # Must either:
        # - return None: continue processing this exception
        # - return a Response object: stops process_exception() chain
        # - return a Request object: stops process_exception() chain
        self.driver.quit()
        self.driver.close()
        pass

    def spider_opened(self, spider):
        spider.logger.info("Spider opened: %s" % spider.name)


    def spider_closed(self, spider):
        # 该函数需要手动添加，作用是关闭浏览器
        self.driver.close()
        spider.logger.info("Spider closed: %s" % spider.name)

# 3. 处理爬虫主要逻辑

import scrapy


class LoginSpider(scrapy.Spider):
    name = "login"
    allowed_domains = ["www.XXX.com"]
    start_urls = ["https://www.XXX.com/system/user"]

    def parse(self, response):
        # 使用XPath选择数据
        result = scrapy.selector.Selector(response)
        items = result.xpath('//tr[@class="el-table__row"]')
        items1 = items.xpath('.//*/div[@class="cell"]')

        for item2 in items1:
            text = item2.get()
            print(text)

        return None

    def start_requests(self):
        ## 直接去登录页面，登录逻辑写在中间件
        return [scrapy.Request(url='https://www.XXX.com/login', callback=self.after_login)]

    def after_login(self, response):
        # 进入相应页面爬取内容
        return [scrapy.Request(url=self.start_urls[0], callback=self.parse)]
